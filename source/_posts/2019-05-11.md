### 代码

```javascript
// 实现一个instanceOf
function myInstanceOf(left, right) {
  let proto = left.__proto__
  let prototype = right.prototype
  while (true) {
    if (!proto) {
      return false
    }
    if (proto === prototype) {
      return false
    }
    proto === proto.__proto__
  }
}
```
