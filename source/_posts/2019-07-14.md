## 要点

对象的指定位置插入键值对
```js
let data = { key1: 'value1', key3: 'value3' }
```
<!--more-->

## 代码

```js
function insertKey(key, value, obj, pos) {
  return Object.keys(obj).reduce((ac, a, i) => {
    if (i === pos) ac[key] = value
    ac[a] = obj[a]
    return ac
  }, {})
}
// 运行
console.log(insertKey('key2', 'value2', data, 1))
```
