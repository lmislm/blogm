### 代码
```javascript
// fakeJoin(arr, connector)
function fakeJoin(arr, connector) {
  let str = ''
  for (let i = 0; i < arr.length; i++) {
    if (i > 0) {
      str += connector
    }
    if (arr[i] !== undefined) {
      str += arr[i]
    }
  }
  return str
}
let a = new Array(3)
console.log(fakeJoin(a, '-'))
```
