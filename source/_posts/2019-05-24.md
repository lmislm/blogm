### 前言

一道构造数据结构的题目。
src: [{id:1}, {id:1}, {id:2}]
tgt: {1: [0,1], 2:[2]}

<!--more-->
### 代码
个人解法。希望以后想出更好的解法。
```javascript
function groupByProp (objArray, prop) {
  return objArray.reduce((acc, cur, index) => {
    let key = cur[prop]
    if (!acc[key]) {
      acc[key] = []
    }
    acc[key].push(index)
    return acc
  }, {})
}
```
